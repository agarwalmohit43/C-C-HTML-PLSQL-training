int binarysearch(int *ptr,int n,int m,int l,int u)
{
    
    
    int mid,c=0;
     
   
   
    if(l<=u)
    {
        mid=(l+u)/2;
        if(ptr[mid]==m)
        {
           c=1;
        }else if(m<ptr[mid])
        {
        return binarysearch(ptr,n,m,mid-1,u);
        }else
        {
        return binarysearch(ptr,n,m,mid+1,u);
        }
    }else
    {
        return c;
    }
    
}
